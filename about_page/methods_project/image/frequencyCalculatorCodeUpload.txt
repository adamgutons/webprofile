 */ This project creates a method for calculating sound frequency (HZ)
public class frequencyCalculator {
    //create main method
    public static void main(String [] args){
  
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Enter wavelength to convert nanometers to meters:");
        
        double wavelengthNm = scanner.nextDouble();
     
        double wavelength = convertWavelengthToNanoMeters(wavelengthNm);
        
        System.out.println("Enter velocity to calculate frequency:");
       
        double velocityNm = scanner.nextDouble();

        double frequency = calcFrequency(velocityNm, wavelength);
       
        System.out.println("Display the frequency is " + frequency + "Hz.");
        
    }//close main method

    
    /**
     * This method converts the user entered nanometers to meters
     * @param wavelengthNm this is the user entered nanometer value
     * @return wavelength this is the value resulting from conversion
     */
    public static double convertWavelengthToNanoMeters(double wavelengthNm){
        double conversionVal = (1/10^9);
        double wavelength = wavelengthNm * conversionVal;
        return wavelength;
    }//close convertwavelength method
    

     /**
      * This method calculates the frequency 
      * @param velocityNm this is the user entered velocity value
      * @param wavelength this is the conversion value from the
      * convertWavelengthToNanoMeters method
      * @return this is the result of the equation, or frequency value
      */
    public static double calcFrequency(double velocityNm, double wavelength){
        double frequency = (velocityNm/wavelength);
        return frequency;
    }//close calcFrequency method
    
}//close class
